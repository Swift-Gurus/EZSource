   let uniqueID: String
   let text: String

   // Defines uniqueness of the model
   func hash(into hasher: inout Hasher) {
       hasher.combine(uniqueID)
   }

   // Defines dynamic context of the model
   static func == (lhs: Self, rhs: Self) -> Bool {
       lhs.text == rhs.text
   }
}
